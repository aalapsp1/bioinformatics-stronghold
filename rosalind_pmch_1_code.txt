# Setting the working directory
import os
os.chdir(r'C:\Users\aalap\Dropbox\Rosalind\Bioinformatics Stronghold\26. Perfect Matchings and RNA Secondary Structures\Datasets')

# Parsing data
RawData = open('rosalind_pmch.txt', 'r')
RNA_String = ''.join(Symbol
                     for Symbol
                     in RawData.read().replace('>Rosalind_', '').replace('\n', '')
                     if Symbol.isalpha())

# Determining total possible number of perfect matchings
A_Count = 0
G_Count = 0
for Nucleotide in RNA_String:
    if Nucleotide is 'A':
        A_Count += 1
    if Nucleotide is 'G':
        G_Count += 1
import math
Perfect_Matchings = (math.factorial(A_Count) * math.factorial(G_Count))

# Generating answer
os.chdir(r'C:\Users\aalap\Dropbox\Rosalind\Bioinformatics Stronghold\26. Perfect Matchings and RNA Secondary Structures\Answer')
Answer = open('26. Answer.txt', 'w')
Answer.write(str(Perfect_Matchings))
Answer.close()
